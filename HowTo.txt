#PRIMERO QUE NADA DEBE INSTALAR:
Python 3.8.2  ->  https://www.python.org/ftp/python/3.8.2/python-3.8.2.exe

# INSTALAR DEPENDENCIAS:

python -m pip install --upgrade pip
pip install django
pip install peewee
pip install django-bootstrap4

# CREAR PROYECTO Y APP: (EN VEZ DE CREARLO, BAJENLO DE GIT)

mkdir C:\PrograMate
cd C:\PrograMate
django-admin startproject Clickrentacar
cd Clickrentacar
python manage.py startapp cracAPP
python manage.py makemigrations
python manage.py migrate

# PARA CORRER EL SERVER: (ABRIR PUERTOS EN FIREWALL SI ES NECESARIO)

python manage.py runserver 127.0.0.1:8000
python manage.py runserver 0.0.0.0:8000


# PARA EFECTUAR LOS CAMBIOS DE LOS MODELS EN LA DB: (CREAR TABLAS)
# primero chequear que no hayan errores: < python manage.py check >

python manage.py makemigrations cracAPP
python manage.py migrate cracAPP
python manage.py sqlmigrate cracAPP 0001

# CREAR DJANGO-ADMIN: (Username: admin, Email: admin@clickrentacar.com, Password: admin)
python manage.py createsuperuser
